function renderShowList() { const container = document.getElementById('showList'); const emptyState = document.getElementById('emptyState'); container.innerHTML = ''; if (state.shows.length === 0) { emptyState.classList.remove('hidden'); return; } else { emptyState.classList.add('hidden'); } const sortedShows = [...state.shows].sort((a, b) => { const ratingA = a.rating !== null && a.rating !== undefined ? a.rating : -1; const ratingB = b.rating !== null && b.rating !== undefined ? b.rating : -1; return ratingB - ratingA; }); sortedShows.forEach((show, index) => { const listItem = document.createElement('div'); listItem.className = 'list-item adding'; listItem.setAttribute('data-show-id', show.id); const ratingClass = getRatingClass(show.rating); const ratingDisplay = show.rating !== null && show.rating !== undefined ? show.rating.toFixed(1) : '--'; const safePosterUrl = optimizeImageUrl(show.posterUrl); const yearRange = show.year && show.endYear && show.endYear !== show.year ? `${show.year} – ${show.endYear}` : (show.year || '—'); const episodesText = show.episodes && show.episodes > 0 ? `${show.episodes} серий` : ''; const card = document.createElement('div'); card.className = 'show-card'; card.addEventListener('click', (e) => { if (e.target.closest('.delete-btn')) return; openModal(show.id); }); // Постер const posterContainer = document.createElement('div'); posterContainer.className = 'poster-container'; const posterImg = document.createElement('img'); posterImg.className = 'w-full h-full object-cover poster-img'; posterImg.loading = 'lazy'; posterImg.src = safePosterUrl; posterImg.onload = function() { this.classList.add('loaded'); this.classList.remove('loading'); }; posterImg.onerror = function() { this.src = 'https://via.placeholder.com/200x280?text=No+Image'; this.classList.add('loaded'); }; posterContainer.appendChild(posterImg); // Контент (середина) const contentDiv = document.createElement('div'); contentDiv.className = 'show-card-content'; const title = document.createElement('h3'); title.className = 'show-title'; title.textContent = show.title; contentDiv.appendChild(title); if (show.originalTitle && show.originalTitle !== show.title && !isRussian(show.title)) { const orig = document.createElement('p'); orig.className = 'original-title'; orig.textContent = show.originalTitle; contentDiv.appendChild(orig); } // ЕДИНЫЙ КОНТЕЙНЕР ДЛЯ ВСЕХ МЕТА-ТЕГОВ const metaContainer = document.createElement('div'); metaContainer.className = 'meta-info'; // Статус (если есть) if (show.status) { const statusSpan = document.createElement('span'); if (show.status === 'Running') { statusSpan.className = 'status-badge status-running'; statusSpan.textContent = '🟢 Продолжается'; } else if (show.status === 'Ended') { statusSpan.className = 'status-badge status-ended'; statusSpan.textContent = '🔴 Завершён'; } if (statusSpan.textContent) { metaContainer.appendChild(statusSpan); } } // Платформа (если есть) if (show.network) { const netSpan = document.createElement('span'); netSpan.className = 'platform-tag'; netSpan.style.cssText = 'font-size:10px;color:#fbbf24;background:rgba(251,191,36,0.15);padding:2px 6px;border-radius:4px;'; netSpan.textContent = `📺 ${show.network}`; metaContainer.appendChild(netSpan); } // Жанры (с переводом) if (show.genres && show.genres.length > 0) { show.genres.slice(0, 3).forEach(g => { const genreSpan = document.createElement('span'); genreSpan.className = 'genre-tag'; genreSpan.textContent = translateGenre(g); metaContainer.appendChild(genreSpan); }); } // Год if (yearRange) { const yearSpan = document.createElement('span'); yearSpan.className = 'meta-tag'; yearSpan.innerHTML = ` ${yearRange}`; metaContainer.appendChild(yearSpan); } // Количество серий if (episodesText) { const epSpan = document.createElement('span'); epSpan.className = 'meta-tag'; epSpan.innerHTML = ` ${episodesText}`; metaContainer.appendChild(epSpan); } // Добавляем контейнер с мета-тегами в contentDiv if (metaContainer.children.length > 0) { contentDiv.appendChild(metaContainer); } // Правая панель (номер + действия) const rightDiv = document.createElement('div'); rightDiv.className = 'show-card-right'; const numberDiv = document.createElement('div'); numberDiv.className = 'show-card-number'; numberDiv.textContent = `#${index + 1}`; const actionsDiv = document.createElement('div'); actionsDiv.className = 'show-card-actions'; const ratingDiv = document.createElement('div'); ratingDiv.className = `rating-badge ${ratingClass}`; ratingDiv.textContent = ratingDisplay; const deleteBtn = document.createElement('button'); deleteBtn.className = 'delete-btn'; deleteBtn.textContent = '✕'; deleteBtn.addEventListener('click', (e) => removeShow(e, show.id)); actionsDiv.appendChild(ratingDiv); actionsDiv.appendChild(deleteBtn); rightDiv.appendChild(numberDiv); rightDiv.appendChild(actionsDiv); card.appendChild(posterContainer); card.appendChild(contentDiv); card.appendChild(rightDiv); listItem.appendChild(card); container.appendChild(listItem); }); }
